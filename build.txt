# build commands for manual build on linux / unix based OSs for this repository

step#1 - prepare build
-> rm -rf public && mkdir public

step#2 - copy
-> cp index.html public/index.html
-> cp -r styles public/styles
-> cp -r sounds public/sounds
-> cp -r images public/images

step#3 - bundle
-> esbuild scripts/main.js --bundle --outfile=public/bundle.js --minify --target=es6 --loader:.html=text --loader:.css=text --sourcemap
